  /* COMANDOS */
  const commands = {
    ls: Object.keys(tools),
    help: [
      "Comandos dispon√≠veis:",
      "ls                 ‚Üí listar tools",
      "open <tool>        ‚Üí abrir tool",
      "help               ‚Üí ajuda",
      "clear              ‚Üí limpar terminal"
    ]
  };
  
  document.querySelectorAll(".tool-card button").forEach(btn => {

  /* ================= CORE ================= */

  function createPrompt() {
    const line = document.createElement("div");
    line.className = "line";

    line.innerHTML = `
      <span class="prompt">weblabx@tools</span>:<span class="path">~</span>$ 
      <span class="input"></span><span class="cursor">‚ñà</span>
    `;

    terminal.appendChild(line);
    inputEl = line.querySelector(".input");
    currentInput = "";

    terminal.scrollTop = terminal.scrollHeight;
  }

  function printOutput(text) {
    const out = document.createElement("div");
    out.className = "output";
    out.textContent = text;
    terminal.appendChild(out);
  }

  function runCommand(cmd) {
    if (!cmd) {
      createPrompt();
      return;
    }

    const parts = cmd.split(" ");
    const base = parts[0];
    const arg = parts.slice(1).join(" ");

    /* clear */
    if (base === "clear") {
      terminal.innerHTML = "";
      createPrompt();
      return;
    }

    /* open <tool> */
    if (base === "open") {
      if (!arg) {
        printOutput("uso: open <tool>");
        createPrompt();
        return;
      }

      if (tools[arg]) {
        printOutput(`a abrir ${arg}‚Ä¶`);
        setTimeout(() => {
          window.location.href = tools[arg];
        }, 600);
        return;
      } else {
        printOutput(`tool n√£o encontrada: ${arg}`);
        createPrompt();
        return;
      }
    }

    /* comandos normais */
    if (commands[base]) {
      commands[base].forEach(line => printOutput(line));
      createPrompt();
      return;
    }

    /* comando inv√°lido */
    printOutput(`comando n√£o encontrado: ${cmd}`);
    createPrompt();
  }

  /* ================= INPUT ================= */

  terminal.addEventListener("keydown", (e) => {

    // bloquear scroll da p√°gina
    if (e.key === " ") {
      e.preventDefault();
    }

    if (!inputEl) return;

    // caracteres normais
    if (e.key.length === 1 && !e.ctrlKey && !e.metaKey) {
      currentInput += e.key;
      inputEl.textContent = currentInput;
    }

    // backspace
    if (e.key === "Backspace") {
      e.preventDefault();
      currentInput = currentInput.slice(0, -1);
      inputEl.textContent = currentInput;
    }

    // enter
    if (e.key === "Enter") {
      e.preventDefault();
      runCommand(currentInput.trim());
    }
  });

  /* ================= INIT ================= */
  createPrompt();

});

const resultsBox = document.getElementById("search-results");

if (searchInput && resultsBox) {
  searchInput.addEventListener("input", () => {
    const query = searchInput.value.toLowerCase().trim();
    resultsBox.innerHTML = "";

    if (!query) {
      resultsBox.style.display = "none";
      return;
    }

    const matches = toolsList.filter(tool =>
      tool.name.toLowerCase().includes(query)
    );

    if (!matches.length) {
      resultsBox.style.display = "none";
      return;
    }

    matches.forEach(tool => {
      const item = document.createElement("div");
      item.className = "search-item";
      item.innerHTML = `
        ${tool.name}
        <span>${tool.desc}</span>
      `;

      item.addEventListener("click", () => {
        window.location.href = tool.url;
      });

      resultsBox.appendChild(item);
    });

    resultsBox.style.display = "block";
  });

  // Enter abre o primeiro resultado
  searchInput.addEventListener("keydown", (e) => {
    if (e.key === "Enter") {
      const first = resultsBox.querySelector(".search-item");
      if (first) first.click();
    }
  });

  // clicar fora fecha
  document.addEventListener("click", (e) => {
    if (!e.target.closest(".hero-search")) {
      resultsBox.style.display = "none";
    }
  });
}
